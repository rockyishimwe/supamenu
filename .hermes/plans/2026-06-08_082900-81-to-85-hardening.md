### Task 2.1: Add GET /api/health

Create `backend/routes/health.js`:
```js
const express = require('express');
const router = express.Router();
const mongoose = require('mongoose');

/**
 * @openapi
 * /health:
 *   get:
 *     tags: [System]
 *     summary: Health check endpoint
 *     responses:
 *       200:
 *         description: Service status
 *         content:
 *           application/json:
 *             schema:
 *               type: object
 *               properties:
 *                 status: { type: string }
 *                 timestamp: { type: string }
 *                 mongodb: { type: string }
 */
router.get('/', (req, res) => {
  const dbState = mongoose.connection.readyState;
  const dbStatus = {
    0: 'disconnected',
    1: 'connected',
    2: 'connecting',
    3: 'disconnecting',
  };
  const healthy = dbState === 1;
  res.status(healthy ? 200 : 503).json({
    status: healthy ? 'ok' : 'degraded',
    timestamp: new Date().toISOString(),
    mongodb: dbStatus[dbState] || 'unknown',
  });
});

module.exports = router;
```

Modify `backend/server.js` — add after other route mounts:

## Phase 3: Backend Tests (Backend Architecture + Testing)

The existing `auth.test.js` has 3 tests that all time out due to `vi.mock` being called inside `it()` blocks (vitest hoisting issue). Fix those, then add new route-level tests.

### Task 3.1: Fix broken auth tests

**Files:** `backend/tests/auth.test.js`

**Core issue:** `vi.mock()` inside `it()` blocks is hoisted to the top of the file but its module-level scope conflicts with the dynamic `import()`. The mocks don't apply in time, causing timeouts.

**Fix:** Move mocks to the top level so they're properly hoisted, then use `vi.mocked()` for per-test overrides.

```js
import { describe, it, expect, vi, beforeEach } from 'vitest';

// Mock logger (top-level = properly hoisted)
vi.mock('../utils/logger', () => ({
  default: { info: vi.fn(), error: vi.fn(), warn: vi.fn() },
}));

// Top-level mocks
const mockUserFindOne = vi.fn();
const mockUserSave = vi.fn();
vi.mock('../models/User', () => ({
  default: { findOne: mockUserFindOne },
}));

vi.mock('../models/Restaurant', () => ({
  default: { findById: vi.fn() },
}));

const mockBcrypt = { genSalt: vi.fn(), hash: vi.fn(), compare: vi.fn() };
vi.mock('bcryptjs', () => mockBcrypt);

const mockJwt = { sign: vi.fn() };
vi.mock('jsonwebtoken', () => mockJwt);

describe('Auth Service', () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it('should hash passwords during registration', async () => {
    // User doesn't exist yet
    mockUserFindOne.mockResolvedValue(null);
    mockBcrypt.genSalt.mockResolvedValue('salt');
    mockBcrypt.hash.mockResolvedValue('hashed_pw');
    mockJwt.sign.mockReturnValue('mock_token');

    const authService = await import('../services/authService');

    const result = await authService.registerUser({
      name: 'Test',
      email: 'test@test.com',
      password: 'password123',
      role: 'customer',
    });

    expect(mockBcrypt.genSalt).toHaveBeenCalled();
    expect(mockBcrypt.hash).toHaveBeenCalledWith('password123', 'salt');
    expect(result.token).toBe('mock_token');
  });

  it('should reject login with wrong password', async () => {
    mockUserFindOne.mockResolvedValue({
      _id: '123',
      name: 'Test',
      email: 'test@test.com',
      role: 'customer',
      password: 'hashed',
    });
    mockBcrypt.compare.mockResolvedValue(false);

    const authService = await import('../services/authService');

    await expect(
      authService.loginUser({ email: 'test@test.com', password: 'wrong' })
    ).rejects.toThrow('Invalid credentials');
  });

  it('should return token on successful login', async () => {
    mockUserFindOne.mockResolvedValue({
      _id: '123',
      name: 'Test',
      email: 'test@test.com',
      role: 'customer',
      password: 'hashed',
      walletBalance: 128.50,
      avatar: '',
      customerDetails: {},
      staffDetails: {},
      ownerDetails: {},
      toObject: () => ({ _id: '123', name: 'Test', email: 'test@test.com', role: 'customer' }),
    });
    mockBcrypt.compare.mockResolvedValue(true);
    mockJwt.sign.mockReturnValue('token123');

    const authService = await import('../services/authService');

    const result = await authService.loginUser({ email: 'test@test.com', password: 'correct' });
    expect(result.token).toBe('token123');
    expect(result.user.email).toBe('test@test.com');
  });
});
```

## Risks & Guardrails

1. **JSX-in-.js**: The esbuild `include: /\.js$/` in vitest config might conflict with plain JS modules. If tests fail, narrow to `include: /__tests__\/.*\.js$/`.
2. **Backend test timeout**: If `vi.mock` at top level + `await import(...)` with `resetModules()` still timeouts, switch to synchronous `require()` in tests instead of dynamic `import()`.
3. **No TypeScript**: All files remain `.js` — vitest handles JSX-in-.js through esbuild config only.
4. **Injecting validation to existing routes**: Adding `body(...), validate` chains to existing route handlers is non-breaking — they reject malformed data but pass valid data through exactly as before. If Swagger docs already specify schemas, no doc updates needed.
